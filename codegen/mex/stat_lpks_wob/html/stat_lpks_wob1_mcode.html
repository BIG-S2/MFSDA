1function [flag,GCVs] = stat_lpks_wob(Coord,Xdesign,Ydesign)
 2%#codegen
 3% stat_lpks_wob is to implement Zhu's (2010) method of local polynomial kernel smoothing (order = 1) for optimal bandwidth selection
 4%
 5% Input:
 6%     Coord      - a L x d matrix of the coordinates of all vertices aligned on the sphere.
 7%     Xdesign    - a n x p0 normalized design matrix.
 8%     Ydesign    - a n x L x m matrix of surface shape data.
 9% Output:
10%     flag       - a 1 x m vector of indice for optimal bandwidth.
11%     GCVs       - a nh x m matrix of GCVs.
12% Reference:
13% Zhu, Hongtu; Li, Runze; Kong, Linglong. Multivariate varying coefficient model for functional responses. Ann. Statist. 40, 2634--2666, 2012.
14% doi:10.1214/12-AOS1045. http://projecteuclid.org/euclid.aos/1359987533.
15% %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
16% Please run
17%    Ydesign = stat_read_vtk(file_dir)
18%    Coord = stat_read_vtk(file_dir)
19%    Xdesign = stat_read_x(design_data, n_Interest, n_Con)
20% before you use stat_lpks_wob
21%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
22% March, 2017 @ Chao Huang
23%
24
25%% SetUp
26n=size(Xdesign,1);     %   n  = sample size    p0  = number of covariates
27[L,d]=size(Coord);    %   L = number of vertices     d  = dimension of corrdinate
28m=size(Ydesign,3);     %   m  = number of shape features (here m=d)
29
30efitYdesign=Ydesign*0;  % fitted Ydesign
31
32
33%% bandwidth SetUp
34nh=50;      % the number of candidate bandwidth
35GCVs=zeros(nh,m);      % GCV performance function
36
37
38%% GCV
39Tmat0=zeros(L,L,d+1);    % L x L x d+1 matrix
40Tmat0(:,:,1)=ones(L);
41
42for dii=1:d   % go through all dimension
43    Tmat0(:,:,dii+1)=Coord(:,dii)*ones(1,L)-ones(L,1)*Coord(:,dii)';
44end
45
46K = 5; % number of floders
47ind=floor(linspace(1,n,K+1));
48
49
50for nhii=1:nh   % go through all candidate bandwidth
51    
52    Kmat=ones(L);
53    
54    for dii=1:d   % go through all dimension
55        Coord_range=2;
56        hmin=0.01;  % minimum bandwidth
57        hmax=0.5*Coord_range;    % maximum bandwidth
58        vh=logspace(log10(hmin),log10(hmax),nh);    % candidate bandwidth
59        h=vh(nhii);
60        Kmat=Kmat.*(real(EpKernel(Tmat0(:,:,dii+1)/h,h)));    % Epanechnikov kernel smoothing function
61    end
62    
63    Tmat=permute(Tmat0,[1,3,2]);   % L x d+1 x L matrix
64    
65    for mii=1:m  % go through all features
66            for Lii=1:L   % go through all vertex
67                KX=Kmat(:,Lii)*ones(1,(d+1)).*Tmat(:,:,Lii);      % L0 x d+1 matrix
68                for kk=1:K
69                    indx=ones(1,n);indx(ind(kk):ind(kk+1))=0;
70                    Xdesign0=Xdesign(indx==1,:);
71                    PX0=(Xdesign0'*Xdesign0)\Xdesign0';
72                    Ydesign0=Ydesign(indx==1,:,:);
73                    efitBetas0=PX0*Ydesign0(:,:,mii)*([1 zeros(1,d)]/(KX'*Tmat(:,:,Lii)+eye(d+1)*0.0001)*KX')';
74                    efitYdesign(ind(kk):ind(kk+1),Lii,mii)=Xdesign(ind(kk):ind(kk+1),:)*efitBetas0;
75                end
76            end
77        GCVs(nhii,mii)=sum(sum((Ydesign(:,:,mii)-efitYdesign(:,:,mii)).^2))/(n*L);
78    end
79
80end
81
82[~,flag]=min(GCVs);   % optimal bandwidth
83%mh=vh(flag);   % optimal bandwidth
84
85end
